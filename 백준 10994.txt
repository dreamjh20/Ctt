#include <stdio.h>

void print(int n, int m){
	for(int i = 0; i <(n-1)*4+1; i++){
		if(i == 0 || m == 0 || i == (n-1)*4 || m == (n-1)*4){
			printf("*");
		}
		else{
			if(m %2 == 1){
				printf(" ");
			}
			
			else{
				if(i %2 == 0) printf("*");
				else printf("-");
			}
		}
	}
	printf("\n");
}

int main(){
	int N;
	scanf("%d", &N);
	for(int i = 0; i<(N-1)*4+1; i++) print(N, i);
	return 0;
}
